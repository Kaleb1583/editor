    <h1>Hello, World!</h1>
    <p>Write some code!</p>
    <button onclick="change(-1)">-</button><a id="middle">0</a><button onclick="change(+1)">+</button>

    <script>
        function change(inde) {
            if(inde == 1) {
                var before = document.getElementById("middle").innerText;
                var after = (parseInt(before)+1);
                document.getElementById("middle").innerText = String(after);
            } else if(inde == -1) {
                var before = document.getElementById("middle").innerText;
                if(parseInt(before) == 0) {return;}
                var after = (parseInt(before)-1);
                document.getElementById("middle").innerText = String(after);
            }
        }
    </script>
</body>

</html>
